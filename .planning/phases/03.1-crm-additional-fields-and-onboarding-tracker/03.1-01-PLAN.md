import { z } from "zod/v4";

export const EmployeeDataSchema = z.object({
  positionNumber: z.string().min(1, "Position number is required"),
  firstName: z.string().min(1, "First name is required"),
  lastName: z.string().min(1, "Last name is required"),
  ssn: z.string().regex(/^\d{3}-\d{2}-\d{4}$/, "SSN must be in XXX-XX-XXXX format").optional(),
  address: z.string().min(1, "Address is required"),
  city: z.string().min(1, "City is required"),
  state: z.string().length(2, "State must be a 2-letter code (e.g., CA)"),
  postalCode: z.string().regex(/^\d{5}(-\d{4})?$/, "Postal code must be XXXXX or XXXXX-XXXX format"),
  wage: z.string().regex(/^\$/, "Wage must start with $ (e.g., $17.75 per hour)"),
  dob: z.string().regex(/^\d{2}\/\d{2}\/\d{4}$/, "Date of birth must be in MM/DD/YYYY format").optional(),
  effectiveDate: z.string().regex(/^\d{2}\/\d{2}\/\d{4}$/, "Effective date must be in MM/DD/YYYY format"),
});

export type EmployeeData = z.infer<typeof EmployeeDataSchema>;
export function validateEmployeeData(raw: Record<string, string | null>): EmployeeData;
